// Vercel serverless function — handles contact & partner form submissions // For email delivery, add RESEND_API_KEY to Vercel Environment Variables (free at resend.com) // Without it, form submissions still return success (no email sent). export default async function handler(req, res) { res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); if (req.method === 'OPTIONS') return res.status(200).end(); if (req.method !== 'POST') return res.status(405).json({ error: 'Method not allowed' }); const { type, name, company, phone, email, message, trade, areas, plan } = req.body; const TO = 'builtnotbrokensolutions.org@gmail.com'; let subject, html; if (type === 'contractor') { subject = `🏗️ New Partner: ${name} — ${trade}`; html = `

New Contractor Registration

Name${name}
Company${company || '—'}
Phone${phone}
Email${email}
Trade${trade}
Areas${areas || '—'}
Plan${plan === 'per-lead' ? '$100 Per Lead' : '8% Revenue Share'}

Built Not Broken Solutions — builtnotbrokensolutions.org

`; } else { subject = `📩 Contact: ${name}`; html = `

New Contact Message

Name${name}
Phone${phone ? `${phone}` : '—'}
Email${email ? `${email}` : '—'}

Message

${(message || '').replace(/\n/g, '
')}

Built Not Broken Solutions — builtnotbrokensolutions.org

`; } const resendKey = process.env.RESEND_API_KEY; if (!resendKey) { // No email configured — still acknowledge receipt return res.status(200).json({ success: true, note: 'received (email not configured)' }); } try { const r = await fetch('https://api.resend.com/emails', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${resendKey}` }, body: JSON.stringify({ from: 'Built Not Broken ', to: TO, reply_to: email || undefined, subject, html }) }); if (r.ok) return res.status(200).json({ success: true }); const err = await r.json(); return res.status(500).json({ error: err.message || 'Email send failed' }); } catch (err) { return res.status(500).json({ error: err.message }); } }
